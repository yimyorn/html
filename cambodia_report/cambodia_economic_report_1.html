របាយការណ៍សេដ្ឋកិច្ចប្រទេសកម្ពុជា

សេដ្ឋកិច្ចប្រទេសកម្ពុជាស្ទុះឡើងយ៉ាងខ្លាំង

ភ្នំពេញ ថ្ងៃទី១៨ ខែឧសភា ឆ្នាំ​២០២៣

សេដ្ឋកិច្ច​របស់​កម្ពុជា​កំពុង​មានសន្ទុះស្វាហាប់​ឆ្ពោះទៅ​ការ​ស្តារ​ឡើង​វិញ​ពេញលេញ ដោយ​សារមាន​​ការ​ជំរុញ​​ពី​ផ្នែកនាំចេញ និង​ពី​កំណើន​ក្នុង​ផ្នែក​សេវា​កម្ម និង​កសិកម្ម។​ធនាគារពិភពលោក​​និយាយ​ដូច្នេះ ​នៅ​ក្នុង​របាយការណ៍បច្ចុប្បន្នភាពអំពី​សេដ្ឋកិច្ច​​កម្ពុជា ដែល​ចេញផ្សាយ​នៅ​ថ្ងៃនេះ។ របាយការណ៍នេះបង្ហាញថា ​ការ​ងើប​ឡើងវិញនៃសេដ្ឋកិច្ច​ក្នុង​ឆ្នាំ​២០២២កន្លងទៅមាន​សភាពរឹងមាំ ដោយក្នុងនោះ​​​សម្រេច​បានកំណើនក្នុង​កម្រិត​​ ៥,២ភាគរយ។ សន្ទុះងើបឡើងវិញក្នុង​ផ្នែកសេវាកម្ម ជា​ពិសេស​ក្នុង​ផ្នែក​ពាណិជ្ជកម្ម, ការ​ធ្វើ​ដំណើរ, និង​បដិសណ្ឋារកិច្ច បាន​នាំកម្ពុជា​ត្រឡប់ទៅរក​កម្រិតកំណើន​ដែលធ្លាប់​​មាន​កាល​ពីមុនពេល​ផ្ទុះ​ជំងឺ​កូវីដ-១៩ ក្នុងនោះការមកដល់នៃភ្ញៀវបរទេសមានការ​កើន​ឡើង​ខ្លាំង​ស្រ​បពេល​ដែល​កម្ពុជា​ធ្វើ​ម្ចាស់​ផ្ទះ​នៃ​ការ​ប្រកួ​តកីឡា​ស៊ីហ្គេម និង​ប៉ារ៉ាហ្គេម​នៃសមាគមប្រជាជាតិ​អាស៊ីអាគ្នេយ៍។ ការ​ស្ទុះ​ងើបឡើង​វិញ​នៃ​សេដ្ឋ​កិច្ច​កម្ពុជា​នេះ​​បាន​មកក៏​ដោយ​សារ​ថ្លៃ​ទំនិញ​ក្នុង​ស្រុកមាន​ស​ភាព​ធូរ​ស្រាល​បន្តិចវិញ ទន្ទឹម​គ្នា​នឹង​ការ​រក្សា​បាន​នូវ​លំនឹង​នៃ​ថ្លៃ​ប្រេងឥន្ធនៈ និង​ទំនិញ​ នៅ​ទូទាំងពិភពលោក​ ផង​ដែរ។ ​វិស័យ​កសិកម្ម​របស់​កម្ពុជា​កំពុង​រីកវិសាលភាពរបស់ខ្លួន ដោយមាន​ច្រក​ចូល​ទៅ​កាន់​ទីផ្សារក្នុងតំបន់​​កាន់​តែ​ច្រើន​បន្ទាប់​ពីការផ្តល់​សច្ចាប័ន​លើកិច្ច​ព្រម​ព្រៀងពាណិជ្ជកម្មសេរី។ នៅក្នុង​បរិបទនេះ ធនាគារពិភពលោក​ព្យាករ​ថា កំណើន​សេដ្ឋកិច្ច​របស់​កម្ពុជានៅ​ក្នុង​ឆ្នាំ​២០២៣ ​នឹង​ស្ទុះឡើង​ដល់​កម្រិត ៥,៥ភាគរយ។ យ៉ាង​ណា​ក៏ដោយ ការ​អូសបន្លាយ​នៃ​ការ​ធ្លាក់​ចុះ​នៃ​តម្រូវ​ការ​នៅ​បរទេស​នឹង​អាចធ្វើ​ឱ្យ​ផ្នែកផលិតកម្ម​សម្រាប់​នាំចេញ​ចុះខ្សោយ ហើយ​ការ​បន្ត​រឹត​បន្តឹង​ផ្នែក​ហិរញ្ញវត្ថុ​លើ​សកលោក ​អាចបង្កហានិភ័យមួយដ​ល់ផ្នែកហិរញ្ញវត្ថុរបស់​កម្ពុជា​ ដែល​មាន​ការ​ពឹង​អាស្រ័យ​ខ្លាំង​លើ​ទុន​មក​ពី​ក្រៅ​សម្រាប់​វិនិយោគ។ ប្រទេសកម្ពុជា នឹងមានការរីកចម្រើននៅខ្ទង់ឆ្នាំ ២០៣០ ឬ ២០៥០ អាស្រ័យទៅលើវិស័យដូចជា វិស័យកសិកម្ម វិស័យឧស្សាហកម្ម វិស័យដឹកជញ្ជួន និង វិស័យឧស្សាហរណកម្ម។

លោកស្រី ម៉ារីយ៉ាម សាលីម នាយកគ្រប់គ្រងនៃធនាគារពិភពលោកប្រចាំនៅកម្ពុជា មាន​ប្រសាសន៍ថា៖


“ដើម្បី​រក្សា​ឱ្យ​បាន​នូវ​ដំណើរងើបឡើងវិញនៃសេដ្ឋកិច្ច កម្ពុជា​ចាំបាច់​ត្រូវ​ធ្វើ​ពិពិធកម្ម​ផលិតផល​ និង​ទី​ដៅ​ទេសចរណ៍​ឱ្យ​កាន់​តែ​សម្បូរ​បែប​ថែមទៀត។ ជាមួយគ្នានេះ កម្ពុជា​ក៏​ចាំបាច់​ត្រូវ​បង្កើន​សមត្ថ​ភាពប្រកួត​ប្រជែង​របស់​ខ្លួន​ដែរ តាម​រយៈ​ការ​ពង្រីក​បណ្តាញផ្លូវគមនាគមន៍, កាត់​បន្ថយ​របាំងពាណិជ្ជកម្ម និង​សម្រួល​នីតិវិធី​​គយ​ឱ្យ​កាន់​តែរលូនថែមទៀត។” សម្រាប់​រយៈពេលមធ្យម សេដ្ឋកិច្ច​ត្រូវ​បាន​រំពឹង​ថា​នឹង​កើន​ដល់​ ៦ភាគរយ ដោយមាន​ការ​ទ្រ​ទ្រង់​ពីសំណាក់​សន្ទុះ​ស្វាហាប់​នៃការនាំចេញទំនិញ និង​សេវាកម្ម និង​ពីសំណាក់​ការ​កើនឡើងនៃការវិនិយោគ ជា​ពិសេស​ក្រោម​រូប​ភាពដៃគូ​រវាង​ផ្នែកសាធារណៈនិងផ្នែកឯកជន ទៅលើ​គម្រោង​ហេដ្ឋា​រចនាសម្ព័ន្ធធំៗ មាន​កំពង់​ផែសមុទ្រ​និង​ផ្លូវជា​ដើម។ របាយការណ៍នេះនិយាយទៀតថា កម្ពុជា​អាច​ជំរុញ​​កំណើន​នេះ​បាន​តាម​រយៈ​ការវិនិយោគលើតភ្ជាប់នៃហេដ្ឋារចនាសម្ព័ន្ធ និងការអភិវឌ្ឍធនធានមនុស្ស ​ធានា​បាន​នូវ​ស្ថិរភាពផ្នែកហិរញ្ញវត្ថុ និង​​ជំរុញ​ការ​ធ្វើ​ពិពិធ​កម្ម​ការនាំចេញ ដើម្បី​ពង្រីក​ភាពធន់របស់​សេដ្ឋកិច្ច និង​ភាព​ប្រកួតប្រជែង​របស់​ខ្លួន។ របាយការណ៍នេះមានផ្នែកមួយ ដែលផ្តោតសំខាន់ទៅលើ​ការ​ពិនិត្យ​លទ្ធភាព​ដែលកម្ពុជា​អាចធ្វើ​បាន​ក្នុ​ងការ​បង្កើន​ការ​ចំណាយ​សាធារណៈ​ក្នុង​ប៉ុន្មាន​ឆ្នាំចុងក្រោយនេះ ជា​ពិសេស​ទៅលើ​វិស័យ​សុខា​ភិបាល​ និង​អប់រំ ប៉ុន្តែ​របាយការណ៍ក៏​ធ្វើ​ការកត់សម្គាល់ដែរថា ចាំបាច់​ត្រូវ​បង្កើន​ការចំណាយនោះ​ថែមទៀត។ ការផ្តោត​ការ​យក​ចិត្ត​ទុក​ដាក់​ទៅលើ​ប្រសិទ្ធផលផ្នែក​បែងចែកនិង​ផ្នែកប្រតិបត្តិការ​របស់​ធន​ធាន អាចជួយ​បាន​ក្នុង​ការ​ឆ្លើយ​តប​ចំពោះ​​បញ្ហា​នៃ​ការដាក់​​ពង្រាយ​មិនស្មើភាពគ្នានៃគ្រូបង្រៀន​​​​ និងចំពោះ​បញ្ហា​នៃ​កង្វះភាពទុកចិត្ត​របស់​សាធារណ​ជន​ទៅលើ​សេវា​សុខាភិបាល​ក្នុង​ស្រុក។ របាយការណ៍បច្ចុប្បន្នភាពសេដ្ឋកិច្ច​ ផ្តោតសំខាន់លើ​ការ​វិវត្ត​នៃ​​សេដ្ឋកិច្ច​ក្នុង​រយៈពេលខ្លី និង​មធ្យម ហើយ​ចេញ​ផ្សាយ​ពីរដង​ក្នុងមួយឆ្នាំ។
